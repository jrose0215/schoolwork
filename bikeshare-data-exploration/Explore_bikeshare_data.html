Question 1¶

What hour of the day has the highest number of bike trips?

In [5]:
# Function to add an hour column (0–23) based on Start.Time
add_hour_column <- function(df) {
  st <- trimws(as.character(df$Start.Time))
  dt <- as.POSIXct(st, format = "%Y-%m-%d %H:%M:%S")
  df$hour <- as.numeric(format(dt, "%H"))
  df
}
